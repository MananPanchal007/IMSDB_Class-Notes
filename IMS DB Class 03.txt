						2. DLI FUNCTION		--> GET THE DATA  -> GU,GN,GHU, GHN , GNHP ,

								LINKAGE SECTION.

									CALL 'CBLTDLI'  USING  1. DLI-FUCNTION
														2. PCB-MASK-1
														3. SEGEMENT-IO-AREA
														4A. SSA		--> EACH SSA IS FOR 1 SEGMENT
														4B. SSA
														4C. SSA.
									GOBACK.

	MAINPGM:   					SUBPGM
								PROCEDURE DIVISION USING VARA,VARB,VARC.
>> 1. CALL 'SUBPGM' USING VARB,VARC,VARA.
>> 2. CALL 'SUBPGM' USING VARA,VARC,VARB.
>> 3. CALL 'SUBPGM' USING VARA,VARB,VARC.	-- ONLY THIS VALID.

CREATE THE DBD
	
			DBD	NAME=DATABASE-NAME,ACCESS=HDAM/HIDAM/HISAM		(HDAM & HIDAM IS HIGHLY USED IN THE PROJECT
			DATASET DD1=DDNAME*
			SEGM		NAME=ORGSEG01,PARENT=0,BYTES=15
			FIELD	NAME=(ORGCODE,SEQ,U),START=1,BYTES=5				03  ORG-CODE   PIC   X(05).	   SSA -> ORGCODE
			FIELD	NAME=ORGNAME,START=6,BYTES=10
			SEGM		NAME=CLIENT05,PARENT=ORGSEG01,BYTES=50			-> NO KEY FIELDS  > LAYOUT -WILL BE COPYBOOK
			SEGM		NAME=PROJECT,PARENT,CLIENT05,BYTES=60				

			DBDGEN
			FINISH
			END
			IN IMS -> KEY FILED / SEQ FIELD -> OPTIONAL 		U - UNIQUE /  M - MULTIPLE

	=> DBD CODE  -> DBDGEN PROCESS -->  LOAD**

==> 
CREATE THE PSB:  

			PCB	TYPE=DB,NAME=DATABASE-NAME,KEYLEN=XX,PROCOPT=?				KEYLEN=10		KEYLEN=05
			SENSEG NAME=ORGSEG01,PARENT=0							EX: BYTES=5		BYTES=5
			SENSEG NAME=CLIENT05,PARENT=ORGSEG01							BYTES=5		NO KEY
			...
			PCB  TYPE=DB,NAME=DATABASE-NAME,KEYLEN=XX,PROCOPT=?
			SEGSEG..
			...
			PSBGEN LANG=COBOL,PSBNAME=NAME**				-- USE THE SAME NAME IN EXECUTION
			END
			

		>> PROCOPT--> TYPE OF ACCESS ON THE PCB-SEGEMENT

				-> GRID ALL O		= PROCOPT=G | GO		-> ON GET CALL ARE ALLOWED > READ  --> GET ONLY -> NO LOCK - PRERFOANCE INCREASDE
								  PROCOPT=I		- ISRT
								 PROCOPT=A  	-> ALL ACCESS
								PROCOPT=GOP		-> GET ONLY + PATH CALL

			G -GET , R-REPLACE , I - INSERT  -D -DELETE
